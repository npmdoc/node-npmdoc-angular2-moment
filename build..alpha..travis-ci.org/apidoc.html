function angular2-moment.TimeAgoPipe.prototype.getSecondsUntilUpdate (momentInstance)

  • description and source-code
    getSecondsUntilUpdate = function (momentInstance) {
    var howOld = Math.abs(momentConstructor().diff(momentInstance, 'minute'));
    if (howOld < 1) {
        return 1;
    }
    else if (howOld < 60) {
        return 30;
    }
    else if (howOld < 180) {
        return 300;
    }
    else {
        return 3600;
    }
}
